fetch_roadmap.py — Sleutelfuncties & Datastructuren
📥 fetch_csv()
# Haalt roadmap CSV op van Microsoft def fetch_csv() → list[dict]: url = "aka.ms/MSRoadmapCSV" headers = { "User-Agent": "Mozilla/5.0", "Accept-Language": "en-US" } raw = curl(url, headers) return csv.DictReader(raw)
🔤 translate_items()
# Vertaalt alleen nieuwe/gewijzigde items def translate_items( items: list[dict], cache: dict ) → list[dict]: translator = GoogleTranslator( source="en", target="nl" ) for item in items: h = hash(item["Title"]) if cache.get(item["id"]) != h: item["title_nl"] = translator .translate(item["Title"])
🔍 detect_product()
# Matcht item op dashboardproduct DETECT_PATTERNS = { "Teams": ["teams", "meeting"], "SharePoint":["sharepoint", "viva"], "Exchange": ["exchange", "outlook"], ... # 16 producten totaal } def detect_product( row: dict ) → str: pf = row["ProductFamily"].lower() return next(k for k,v in DETECT_PATTERNS if pf in v)
data.json — Schema
📦 RoadmapItem (één item in items[])
{ "id": string # Microsoft ID (uniek) "title_nl": string # Vertaalde titel "benefit_nl": string # Vertaalde beschrijving "title_en": string # Originele Engelse titel "product": string # bijv. "Teams", "SharePoint" "status": enum # "uitgerold" | "in_ontwikkeling" "actionType": enum # "auto" | "it" | "medewerker" "rolloutDate": string # ISO-datum bijv. "2024-Q2" "addedDate": string # Datum toegevoegd aan roadmap "modifiedDate": string # Datum laatste wijziging "moreInfoLink": string # URL naar Microsoft docs }

📡 loadData()
# SessionStorage-cache 30 min async function loadData(): cached = sessionStorage.get( "roadmap_data") if cached && !isExpired(cached): return JSON.parse(cached) url = "data.json?v="+Date.now() data = await fetch(url, { signal: AbortSignal.timeout(12000) }) sessionStorage.set("roadmap_data", data)

# Combineert alle actieve filters function applyFilters(items): return items .filter(i => matchProduct(i)) .filter(i => matchAction(i)) .filter(i => matchStatus(i)) .filter(i => matchWeek(i)) .filter(i => matchSearch(i)) .sort(by(activeSort)) # Weekfilter: ISO-kalenderweek (ma-zo) function matchWeek(item): return getISOWeek(item.addedDate) === currentWeek
